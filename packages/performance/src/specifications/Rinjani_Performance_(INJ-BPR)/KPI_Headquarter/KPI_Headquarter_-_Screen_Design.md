## Component Specifications

### Completion Progress Component

```tsx
interface CompletionProgress {
  phase: string;
  period: { start: string; end: string };
  completion_percentage: number;
  status: 'completed' | 'in_progress' | 'upcoming';
  is_locked: boolean;
}
```

### Rating Distribution Chart Component

```tsx
interface RatingDistribution {
  labels: string[];
  actual_values: number[];
  target_values: number[];
  colors: string[];
  show_variance: boolean;
}
```

### Company Metrics Row Component

```tsx
interface CompanyMetricsRow {
  company_name: string;
  goal_setting_completion: number;
  check_in_completion: number;
  average_pi_score: number;
  status: 'on_track' | 'at_risk' | 'behind';
  onClick: () => void;
}
```
